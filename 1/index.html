int n,d[1010],s[200],k=0,sum=0,base=10000;
struct num
{
    int len,a[10010];
    num()
    {
        len=1;
        memset(a,0,sizeof(a));
        a[0]=1;
    }
    int& operator[](int x)
    {
        return a[x];
    }
    void print()
    {
        printf("%d",a[len-1]);
        for(int i=len-2;i>=0;i--)
        printf("%04d",a[i]);
        puts("");
    }
};

num operator*(num x,int y)
{
    for(int i=0;i<x.len;i++)
    x[i]*=y;
    for(int i=0;i<x.len;i++)
    {
        x[i+1]+=x[i]/base;
        x[i]%=base;
    }
    while(x[x.len])
    {
        x[x.len]+=x[x.len-1]/base;
        x[x.len-1]%=base;
        x.len++;
    }
    return x;
}
void wk(int x,int y)
{
    for(int j=0;j<168;j++)
    {
        while(x%prime[j]==0)
        {
            x/=prime[j];
            s[j]+=y;
        }
    }
}
