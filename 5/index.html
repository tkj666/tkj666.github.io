{
    int *hh=new int[K+1];

    hh[i]=0;
    for(int i=0;i<n;i++)
    hh[s[a[i]]]++;
    for(int i=1;i<=K;i++)
    hh[i]+=hh[i-1];
    for(int i=n-1;i>=0;i--)
    b[--hh[s[a[i]]]]=a[i];
}
