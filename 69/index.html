题意：现有一个n行m列的棋盘，一只马欲从棋盘的左上角跳到右下角。每一步它向右跳奇数列，且跳到本行或相邻行。试求跳法种数mod 30011。
题解：DP+矩阵乘法+快速幂
设$f[i][j]$表示走到第$2i-1$列，第$j$行的方案数，$g[i][j]$表示走到第$2i$列，第j行的方案数。那么

设$F[i][j]=\sum_{k=1}^if[k][j],G[i][j]=\sum_{k=1}^ig[k][j]$那么

矩阵推一推就出来了。

#include<bits/stdc++.h>
using namespace std;

struct matrix
{
    int x,y,a[110][110];
    matrix()
    {
        memset(a,0,sizeof(a));
        x=y=0;
    }
    int* operator[](int x)
    {
        return a[x];
    }
}a,b,one;
int mod=30011,n,m;

matrix operator*(matrix x,matrix y)
{
    matrix ans;
    ans.x=x.x;
    ans.y=y.y;
    for(int i=1;i<=ans.x;i++)
    for(int j=1;j<=ans.y;j++)
    for(int k=1;k<=x.y;k++)
    ans[i][j]=(ans[i][j]+x[i][k]*y[k][j])%mod;
    return ans;
}
matrix pow(matrix x,int y)
{
    if(y==0)
    return one;
    else if(y==1)
    return x;
    matrix hh=pow(x,y>>1);
    hh=hh*hh;
    if(y&1)
    hh=hh*x;
    return hh;
}
