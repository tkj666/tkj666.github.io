题解：数位DP
$f[i][j][k][l][o][p]$表示前i个位置是j，有连续k个j，l表示是否满足了连续3个的条件，o表示4和8的状态，0是都没有，1是有4,2是有8，p表示是否贴着上限。

{
    memset(f,0,sizeof(f));

    for(int o=0;o<3;o++)//8和4的状态 
    for(int p=0;p<2;p++)//是否贴着上限 
    {
        int lim=p?s[i+1]-'0':9;/*
        if(f[i][j][k][l][o][p])

            int hh=(u==j)?min(k+1,3):1,oo=(o|((u%4==0)?u/4:0));//hh:连续 oo:8和4的状态

{
    bool eight=0,four=0;
    int con=0,mx=0;
    for(int i=0;i<11;i++)
    {
        if(i==0||s[i]!=s[i-1])
        {
            mx=max(mx,con);
            con=1;
        }
        else
        con++;

}
